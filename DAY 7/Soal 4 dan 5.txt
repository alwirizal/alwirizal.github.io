SOAL 4 :

var flag = 5
var pangkat = 5
var jumlah = 1;

while(flag > 0) {
  jumlah *= pangkat
  flag--
  console.log(jumlah)
}

var pangkat = 6

  jumlah *= pangkat
  console.log(jumlah)
